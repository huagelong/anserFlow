const { execSync } = require('child_process'); const path = require('path');  const platformMap = {   'darwin-x64':   'flowcode-darwin-amd64',   'darwin-arm64': 'flowcode-darwin-arm64',   'linux-x64':    'flowcode-linux-amd64',   'linux-arm64':  'flowcode-linux-arm64',   'win32-x64':    'flowcode-windows-amd64.exe', };  const key = `${os.platform()}-${os.arch()}`; const binary = platformMap[key]; if (!binary) {   console.error(`Unsupported platform: ${key}`);   process.exit(1); }  const version = require('./package.json').version; const url = `https://github.com/flowcode/flowcode/releases/download/v${version}/${binary}`; const dest = path.join(__dirname, 'flowcode');  console.log(`Downloading flowcode ${version} for ${key}...`); execSync(`curl -fsSL ${url} -o ${dest} && chmod +x ${dest}`, { stdio: 'inherit' }); ```  > **发布流程**：`go build` → 多平台二进制 → GitHub Release → `npm publish`。用户 `npm install -g` 时自动从 GitHub Releases 下载对应平台的二进制。
