CREATE INDEX idx_issue_requirement ON issues(requirement_id); CREATE INDEX idx_issue_priority ON issues(project_id, priority); CREATE INDEX idx_issue_category ON issues(project_id, category);  -- ExecutionLog 索引 CREATE INDEX idx_log_issue ON execution_logs(issue_id, attempt);  -- MonitorEvent 去重索引 CREATE INDEX idx_monitor_dedup ON monitor_events(project_id, dedup_hash, created_at); CREATE INDEX idx_monitor_type ON monitor_events(project_id, event_type, created_at);  -- Requirement 索引 CREATE INDEX idx_req_project_status ON requirements(project_id, status);  -- TestCase 索引 CREATE INDEX idx_test_issue ON test_cases(issue_id, status); CREATE INDEX idx_test_review ON test_cases(review_status, created_at);  -- HistoryRecord 索引 (时间线查询 + 实体对比) CREATE INDEX idx_history_target ON history_records(target_type, target_id, created_at DESC); CREATE INDEX idx_history_changed_by ON history_records(changed_by, created_at DESC);  -- Project 索引 CREATE INDEX idx_project_org ON projects(org_id, slug);  -- GitCredential 索引 CREATE INDEX idx_gitcred_project ON git_credentials(project_id, auth_method);  -- LoginLog 索引 CREATE INDEX idx_login_user ON login_logs(user_id, created_at DESC); CREATE INDEX idx_login_status ON login_logs(status, created_at DESC); CREATE INDEX idx_login_ip ON login_logs(ip, created_at DESC);  -- AuditLog 索引 CREATE INDEX idx_audit_user ON audit_logs(user_id, created_at DESC); CREATE INDEX idx_audit_org ON audit_logs(org_id, created_at DESC); CREATE INDEX idx_audit_project ON audit_logs(project_id, created_at DESC); CREATE INDEX idx_audit_action ON audit_logs(action, created_at DESC); CREATE INDEX idx_audit_target ON audit_logs(target_type, target_id, created_at DESC); CREATE INDEX idx_audit_source ON audit_logs(source, created_at DESC); CREATE INDEX idx_audit_created ON audit_logs(created_at DESC);  -- 登录异常检测查询（近1小时同IP失败次数） CREATE INDEX idx_login_bruteforce ON login_logs(ip, status, created_at);  -- Device 索引 CREATE INDEX idx_device_user ON devices(user_id, current);  -- TestResult 索引 CREATE INDEX idx_tr_testcase ON test_results(test_case_id, created_at DESC); ```
