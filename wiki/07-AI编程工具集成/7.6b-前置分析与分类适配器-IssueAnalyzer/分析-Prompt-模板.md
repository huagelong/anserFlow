Git 集成方案](../../08-Git集成方案/README.md) | [10 - 部署与运维](../../10-部署与运维/README.md) | [11 - 测试策略](../../11-测试策略/README.md) ### 分析 Prompt 模板  ```go // internal/adapter/ai_tool/analyze.go  type IssueAnalyzer struct {     opencodePath string }  // AnalyzeInput 分析请求 type AnalyzeInput struct {     Title       string `json:"title"`     Description string `json:"description"`     ProjectName string `json:"project_name"` }  // AnalyzeResult 合并分析+分类结果 type AnalyzeResult struct {     // 前置分析     IsBug           bool   `json:"isBug"`     NeedsCodeChange bool   `json:"needsCodeChange"`     AnalysisReason  string `json:"analysisReason"`     // 分类     Category        string  `json:"category"`   // feature/bug/refactor/docs/infra/test     Confidence      float64 `json:"confidence"` // 0.0 ~ 1.0     CategoryReason  string  `json:"categoryReason"` }  type PlanResult struct {     Summary      string   `json:"summary"`     Detail       string   `json:"detail"`     Steps        []string `json:"steps"`     TestStrategy string   `json:"testStrategy"` }  func (a *IssueAnalyzer) Analyze(ctx context.Context, input AnalyzeInput) (*AnalyzeResult, error) {     // 构建分析+分类 prompt（一次调用完成两项任务）     prompt := fmt.Sprintf(`你是 Issue 分析器。对给定 Issue 同时完成两项任务：  任务 1 — 前置分析： - isBug: 这是否是一个 bug / 缺陷 / 异常报告？（true/false） - needsCodeChange: 解决此 Issue 是否需要修改源代码？（true/false）   · needsCodeChange=false 的场景：纯文档更新、配置变更（无需代码）、     问题澄清/讨论、纯部署操作等 - analysisReason: 一句话说明判断依据  任务 2 — 分类： 将 Issue 分到以下类别之一： - feature: 新功能、新增特性 - bug: 错误修复、异常处理 - refactor: 代码重构、性能优化 - docs: 文档、注释 - infra: 部署、CI/CD、环境配置、依赖升级 - test: 测试用例、测试相关  输出严格 JSON 格式： {"isBug":false,"needsCodeChange":true,"analysisReason":"...",  "category":"feature","confidence":0.95,"categoryReason":"..."}  项目: %s 标题: %s 描述: %s`, input.ProjectName, input.Title, truncate(input.Description, 2000))      // 调用 opencode     cmd := exec.CommandContext(ctx, "opencode",         "run",         "--input", prompt,         "--model", "claude-haiku",         "--max-turns", "1",         "--output-format", "json",     )      output, err := cmd.CombinedOutput()     if err != nil {         return nil, fmt.Errorf("analyze failed: %w, output: %s", err, string(output))     }      var result AnalyzeResult     if err := json.Unmarshal(extractJSON(string(output)), &result); err != nil {         return nil, fmt.Errorf("parse analyze result: %w", err)     }      // 校验     categories := map[string]bool{         "feature": true, "bug": true, "refactor": true,         "docs": true, "infra": true, "test": true,     }     if !categories[result.Category] {         return nil, fmt.Errorf("invalid category: %s", result.Category)     }      return &result, nil } ```
