config/sandbox.yaml sandbox:   # ── 全局默认值 ──   image: "flowcode/sandbox:latest"     # 可改为私有镜像仓库地址   memory_mb: 2048                       # 内存限制 (MB)   cpus: 2                               # CPU 核数   disk_mb: 2048                         # 磁盘限制 (MB)   timeout_seconds: 600                  # 执行超时 (秒)   keep_on_error: true                   # 失败保留现场   network: "bridge"                     # bridge / host / none    # ── 分布式节点池 ──   nodes:     - "unix:///var/run/docker.sock"    # 本机 dockerd     # - "tcp://192.168.1.10:2375"      # 节点 1     # - "tcp://192.168.1.11:2375"      # 节点 2     # - "tcp://10.0.0.5:2375"          # 节点 3    # 节点调度策略   schedule: "least_containers"          # least_containers | round_robin | random    # ── 项目级覆盖 (按 project_id) ──   overrides:     "proj-abc123":                      # 大型项目，给更多资源       memory_mb: 4096       cpus: 4       disk_mb: 4096       image: "flowcode/sandbox:go-1.22" # 专用镜像     "proj-def456":                      # 轻量项目，节省资源       memory_mb: 1024       cpus: 1 ```  #### Go 配置加载  ```go // internal/service/sandbox/config.go  type SandboxGlobalConfig struct {     Image          string            `yaml:"image"`     MemoryMB       int64             `yaml:"memory_mb"`     CPUs           int64             `yaml:"cpus"`     DiskMB         int64             `yaml:"disk_mb"`     TimeoutSeconds int               `yaml:"timeout_seconds"`     KeepOnError    bool              `yaml:"keep_on_error"`     Network        string            `yaml:"network"`     Nodes          []string          `yaml:"nodes"`     Schedule       string            `yaml:"schedule"`     Overrides      map[string]OverrideConfig `yaml:"overrides"` }  type OverrideConfig struct {     Image    *string `yaml:"image"`     MemoryMB *int64  `yaml:"memory_mb"`     CPUs     *int64  `yaml:"cpus"`     DiskMB   *int64  `yaml:"disk_mb"` }  // Resolve 合并全局 + 项目覆盖 func (c *SandboxGlobalConfig) Resolve(projectID string) SandboxConfig {     cfg := SandboxConfig{         Image:    c.Image,         MemoryMB: c.MemoryMB,         CPUs:     c.CPUs,         DiskMB:   c.DiskMB,     }     if ov, ok := c.Overrides[projectID]; ok {         if ov.Image != nil    { cfg.Image = *ov.Image }         if ov.MemoryMB != nil { cfg.MemoryMB = *ov.MemoryMB }         if ov.CPUs != nil     { cfg.CPUs = *ov.CPUs }         if ov.DiskMB != nil   { cfg.DiskMB = *ov.DiskMB }     }     return cfg } ```  #### 环境变量覆盖  ```bash # 通过环境变量快速覆盖，优先级高于 YAML SANDBOX_IMAGE=my-registry/sandbox:v2 SANDBOX_MEMORY_MB=4096 SANDBOX_CPUS=4 SANDBOX_DISK_MB=4096 SANDBOX_TIMEOUT=900 SANDBOX_NODES="unix:///var/run/docker.sock,tcp://node1:2375,tcp://node2:2375" ```  #### DockerSandbox 工厂方法  ```go func NewDockerSandbox(cfg *SandboxGlobalConfig) *DockerSandbox {     clients := makeClientPool(cfg.Nodes)     return &DockerSandbox{         clients:  clients,         cfg:      cfg,         schedule: newScheduler(cfg.Schedule),     } }  // Create 根据项目 ID 选择节点 + 解析配置 func (d *DockerSandbox) Create(ctx context.Context, projectID, issueID string) (*Sandbox, error) {     cfg := d.cfg.Resolve(projectID)          // 合并项目覆盖     cli := d.schedule.Select(d.clients)      // 选择节点     // ... 用 cfg + cli 创建容器 } ```
