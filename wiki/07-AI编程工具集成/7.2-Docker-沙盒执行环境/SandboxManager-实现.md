        &container.HostConfig{             Mounts: []mount.Mount{                 {                     Type:   mount.TypeVolume,                     Source: volumeName,                     Target: sb.WorkDir,     }      execResp, err := d.cli.ContainerExecCreate(ctx, sb.ContainerID, execConfig)     if err != nil {         return nil, fmt.Errorf("exec create: %w", err)     }      attachResp, err scanner := bufio.NewScanner(attachResp.Reader)     var fullOutput strings.Builder     for scanner.Scan() {         line := scanner.Text()         fullOutput.WriteString(line + "\n")         if onLog != nil {             onLog(line)         }     }      // 获取退出码     inspectResp, _ := d.cli.ContainerExecInspect(ctx, execResp.ID)      return &ExecResult{         Output:   fullOutput.String(),         ExitCode: inspectResp.ExitCode,     }, nil }  // Cleanup 清理沙盒 func (d *DockerSandbox) Cleanup(ctx context.Context, sb *Sandbox) error {     // 停止容器     timeout := 10     d.cli.ContainerStop(ctx, sb.ContainerID, container.StopOptions{Timeout: &timeout})     // 删除卷     d.cli.VolumeRemove(ctx, sb.VolumeName, true)     return nil } ```
