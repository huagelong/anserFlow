部署与运维](../../10-部署与运维/README.md) | [11 - 测试策略](../../11-测试策略/README.md) ### 适配器实现  ```go // internal/adapter/ai_tool/decompose.go  type RequirementDecomposer struct {     opencodePath string }  // DecomposeResult 拆解结果 type DecomposeResult struct {     Issues []DecomposeIssue `json:"issues"`     Summary string          `json:"summary"` // 拆解思路概述 }  type DecomposeIssue struct {     Title       string `json:"title"`       // Issue 标题     Description string `json:"description"` // Issue 描述     Priority    string `json:"priority"`    // p0 / p1 / p2 / p3     Category    string `json:"category"`    // bug/feature/refactor/docs/infra/test     Reason      string `json:"reason"`      // 拆解理由     AIToolHint  string `json:"aiToolHint"`  // 推荐 AI 工具 (claude-code/codex/opencode) }  func (d *RequirementDecomposer) Decompose(ctx context.Context, input DecomposeInput) (*DecomposeResult, error) {     prompt := fmt.Sprintf(`你是需求拆解专家。将以下需求拆解为 2~6 个可独立执行的子 Issue。  拆解原则： - 每个 Issue 应是一个独立可完成的任务单元 - Issue 之间尽量解耦，避免循环依赖 - 优先级：p0(核心阻塞) > p1(本迭代必须) > p2(可顺延) > p3(锦上添花) - 分类：feature(新功能) / bug(缺陷) / refactor(重构) / docs(文档) / infra(基础设施) / test(测试) - aiToolHint：根据任务特征推荐 (opencode: 通用 / codex: OpenAI 生态 / claude-code: 复杂逻辑)  输出严格 JSON 格式： {   "summary": "拆解思路概述",   "issues": [     {"title":"...","description":"...","priority":"p0",      "category":"feature","reason":"...","aiToolHint":"claude-code"}   ] }  项目: %s 需求标题: %s 需求描述: %s 技术栈: %s`,         input.ProjectName, input.Title, truncate(input.Description, 3000),         strings.Join(input.TechStack, ", "))      cmd := exec.CommandContext(ctx, "opencode",         "run", "--input", prompt,         "--model", "claude-sonnet-4-20250514",         "--max-turns", "1",         "--output-format", "json",     )      output, err := cmd.CombinedOutput()     if err != nil {         return nil, fmt.Errorf("decompose failed: %w, output: %s", err, string(output))     }      var result DecomposeResult     if err := json.Unmarshal(extractJSON(string(output)), &result); err != nil {         return nil, fmt.Errorf("parse decompose result: %w", err)     }      // 校验：至少 1 个 Issue，最多 10 个     if len(result.Issues) == 0 || len(result.Issues) > 10 {         return nil, fmt.Errorf("invalid issue count: %d", len(result.Issues))     }      return &result, nil } ```
