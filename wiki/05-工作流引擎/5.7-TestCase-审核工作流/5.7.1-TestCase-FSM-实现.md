`reviewStatus`：审核视角，记录当前审核结论 (draft → pending_review → approved / rejected) > - `status`：完整生命周期状态，覆盖审核与执行阶段 (draft → pending_review → approved → generating → generated → passed / failed / rejected) >  > FSM 使用 `status` 字段初始化，因为它覆盖了完整生命周期。审核状态变更通过回调同步更新 `reviewStatus` 字段。  ```go // internal/fsm/test_case_fsm.go  type TestCaseFSM struct {     *fsm.FSM     db  *gorm.DB     svc *FSMDependencies }  func NewTestCaseFSM(tc *model.TestCase, deps *FSMDependencies) *TestCaseFSM {     f := &TestCaseFSM{db: deps.DB, svc: deps}      f.FSM = fsm.NewFSM(         tc.Status,  // 使用 status 字段初始化，覆盖完整生命周期         fsm.Events{             // 审核流程: 同时更新 reviewStatus 和 status             {Name: "submit",    Src: []string{"draft"},             Dst: "pending_review"},             {Name: "approve",   Src: []string{"pending_review"},    Dst: "approved"},             {Name: "reject",    Src: []string{"pending_review"},    Dst: "rejected"},                          // 执行流程: 只更新 status             {Name: "generate",  Src: []string{"approved"},          Dst: "generating"},             {Name: "generated", Src: []string{"generating"},        Dst: "generated"},             {Name: "pass",      Src: []string{"generated"},         Dst: "passed"},             {Name: "fail",      Src: []string{"generated"},         Dst: "failed"},                          // 重试: 从 failed/rejected 回到 approved             {Name: "retry",     Src: []string{"failed", "rejected"}, Dst: "approved"},         },         fsm.Callbacks{             "before_approve": f.beforeTestCaseApprove,             "after_approve":  f.afterTestCaseApprove,   // 自动触发 AI 生成             "after_reject":   f.afterTestCaseReject,             "after_generate": f.afterGenerate,             "after_generated": f.afterGenerated,             "after_pass":     f.afterPassOrFail,             "after_fail":     f.afterPassOrFail,             "after_event":    f.afterTestCaseEvent,      // 通用: 记录历史 + emit hook         },     )     return f } ```
