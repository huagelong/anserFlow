deps}      f.FSM = fsm.NewFSM(         issue.Status,  // 从 DB 恢复当前状态         fsm.Events{             //    事件名        源状态                          目标状态             {Name: "approve",   Src: []string{"draft"},         Dst: "approved"},             {Name: "reject",    Src: []string{"draft"},         Dst: "rejected"},             {Name: "cancel",    Src: []string{"draft", "queued", "running", "paused"}, Dst: "cancelled"},              // 执行链（必须 approved 后才能进入；正式执行前先确保已有计划）             {Name: "enqueue",   Src: []string{"approved"},      Dst: "queued"},             {Name: "pickup",    Src: []string{"queued"},        Dst: "running"},             {Name: "pause",     Src: []string{"running"},       Dst: "paused"},             {Name: "resume",    Src: []string{"paused"},        Dst: "running"},             {Name: "done",      Src: []string{"running"},       Dst: "done"},             {Name: "fail",      Src: []string{"running", "pr_ready"}, Dst: "failed"},              // 跳过执行（opencode 判定 needsCodeChange=false，审核后直接标记完成）             {Name: "skip",      Src: []string{"approved"},      Dst: "done"},              // 重试 & PR             {Name: "retry",     Src: []string{"failed", "cancelled", "done"}, Dst: "approved"},             {Name: "mark_pr_ready", Src: []string{"done"},     Dst: "pr_ready"},             {Name: "create_pr", Src: []string{"pr_ready"},     Dst: "pr_created"},             {Name: "deploy",    Src: []string{"pr_created"},   Dst: "deployed"},              // 关闭 ↔ 重新打开（可重复）             {Name: "close",     Src: []string{"deployed", "done", "cancelled", "failed", "rejected"}, Dst: "closed"},             {Name: "reopen",    Src: []string{"rejected", "cancelled", "closed", "done", "deployed", "pr_created"}, Dst: "reopened"},             {Name: "reopen_to_approved", Src: []string{"reopened"}, Dst: "approved"},              // 快捷通道：管理员跳过审核直接执行（需 Admin 角色 + 显式传参 force=true）             {Name: "fast_exec",  Src: []string{"approved"},    Dst: "queued"},         },         fsm.Callbacks{             // ---- Issue 生命周期回调 ----             "before_approve":          f.beforeApprove,             "after_approve":           f.afterApprove,             "after_reject":            f.afterReject,             "before_enqueue":          f.beforeEnqueue,             "after_enqueue":           f.afterEnqueue,             "before_pickup":           f.beforePickup,             "after_pickup":            f.afterPickup,             "after_done":              f.afterDone,             "after_fail":              f.afterFail,             "after_retry":             f.afterRetry,             "after_cancel":            f.afterCancel,             "before_create_pr":        f.beforeCreatePR,             "after_create_pr":         f.afterCreatePR,             "after_deploy":            f.afterDeploy,             "after_close":             f.afterClose,             "after_reopen":            f.afterReopen,             "after_reopen_to_approved": f.afterReopenToApproved,             "after_skip":              f.afterSkip,      // 跳过执行回调（发射 event + 审计）              // ---- 通用回调 ----             "after_event":             f.afterAnyEvent,     // 所有事件完成后触发             "leave_state":             f.leaveState,        // 离开任意状态             "enter_state":             f.enterState,        // 进入任意状态         },     )     return f } ```
