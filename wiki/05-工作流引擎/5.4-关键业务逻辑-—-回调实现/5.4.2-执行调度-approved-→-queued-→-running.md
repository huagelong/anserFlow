e.Args[0].(*model.Issue)      // 校验前置分析     if !issue.AnalysisCompleted {         e.Cancel(fmt.Errorf("issue analysis not completed"))         return     }      // 校验 AI 工具配置     tool, err := f.svc.DB.GetAIToolConfig(ctx, issue.ProjectID, issue.AITool)     if err != nil || !tool.Enabled {         e.Cancel(fmt.Errorf("AI tool not configured: %s", issue.AITool))         return     }     e.Args = append(e.Args, tool) }  // afterEnqueue — 投递异步任务 func (f *IssueFSM) afterEnqueue(ctx context.Context, e *fsm.Event) {     issue := e.Args[0].(*model.Issue)      // 创建执行日志     log := &model.ExecutionLog{         IssueID: issue.ID,         Status:  "queued",         Attempt: issue.Attempt + 1,     }     f.db.Create(log)      // 投递 Asynq 任务     task, _ := asynq.NewTask("issue:execute", marshal(ExecutePayload{         IssueID: issue.ID,         LogID:   log.ID,     }), asynq.MaxRetry(0))     f.svc.AsynqClient.Enqueue(task)      f.svc.HookRegistry.Emit(ctx, "issue:queued", issue) }  // beforePickup — Worker 拾取任务时触发 func (f *IssueFSM) beforePickup(ctx context.Context, e *fsm.Event) {     // 可选：检查并发数限制     f.svc.HookRegistry.Emit(ctx, "issue:execution:start", e.Args[0]) } ```  Worker 侧的状态驱动 (Asynq handler)：  ```go // internal/queue/handler.go  func ExecuteIssueHandler(ctx context.Context, t *asynq.Task)  if issue.AnalysisCompleted && !issue.NeedsCodeChange {         // 无需代码变更（纯文档/配置/沟通类 Issue），直接标记完成         slog.Info("Issue needs no code change, auto-completing",             "issue", issue.ID, "analysis", issue.AnalysisReason)         fsm.Transition(ctx, issue, "skip")         fsm.SaveState(ctx, issue)         return nil     }      // 3. 生成执行计划（若尚未生成）     if !issue.PlanCompleted {         plan, err := deps.AIToolService.Plan(ctx, issue)         if err != nil {             return err         }         issue.PlanSummary = plan.Summary         issue.PlanDetail = plan.Detail         issue.PlanCompleted = true         issue.PlanGeneratedAt = time.Now()         db.Save(issue)         deps.HookRegistry.Emit(ctx, HookIssuePlanned, issue)     }      // 4. 入队 → 执行中     fsm.Transition(ctx, issue, "pickup")      // 5. 调用 AI 工具     result, err := deps.AIToolService.Execute(ctx, issue)      // 5. 根据结果驱动状态     if err != nil {         fsm.Event(ctx, "fail")     } else {         fsm.Event(ctx, "done")     }     fsm.SaveState(ctx, issue)     return nil } ```
