Skill YAML 前置元数据必须合法     if !strings.HasPrefix(strings.TrimSpace(content), "---") {         return fmt.Errorf("skill must have YAML front matter")     }      return nil } ```  #### Skill 签名验证  组织级 Skill 创建时自动签名，安装公共 Skill 时验证签名来源：  ```go // internal/service/skill_signer.go  type SkillSigner struct {     orgPrivateKey *rsa.PrivateKey  // 组织 RSA 私钥     publicKeys    map[string]*rsa.PublicKey // 受信任发布者公钥 }  // Sign 对 Skill 内容生成 Ed25519 签名 func (s *SkillSigner) Sign(content string) (string, error) {     hash := sha256.Sum256([]byte(content))     sig, err := rsa.SignPKCS1v15(rand.Reader, s.orgPrivateKey, crypto.SHA256, hash[:])     if err != nil {         return "", err     }     return base64.StdEncoding.EncodeToString(sig), nil }  // Verify 验证 Skill 签名，确保内容未被篡改 func (s *SkillSigner) Verify(content, signature string, publisherID string) error {     pubKey, ok := s.publicKeys[publisherID]     if !ok {         return fmt.Errorf("untrusted publisher: %s", publisherID)     }     hash := sha256.Sum256([]byte(content))     sig, _ := base64.StdEncoding.DecodeString(signature)     return rsa.VerifyPKCS1v15(pubKey, crypto.SHA256, hash[:], sig) } ```  #### Skill 信任等级  | 等级 | 来源 | 安全审查 | 签名验证 | 自动注入 | |------|------|:---:|:---:|:---:| | **Trusted** | 组织内部创建 | — | ✅ 组织签名 | ✅ 自动 | | **Verified** | 官方/认证发布者 | ✅ | ✅ 发布者签名 | ✅ | | **Community** | 社区市场 | ❌ | ✅ 基本签名 | ⚠️ 用户确认 | | **Untrusted** | URL 导入 | ❌ | ❌ | ❌ 需手动审核 |  ```go // 安装 Skill 时的安全流程 func (s *SkillService) InstallSkill(ctx context.Context, orgID, skillID string) error {     skill, _ := s.repo.FindByID(skillID)      switch skill.TrustLevel {     case "untrusted":         // 必须管理员手动审核         if !s.isAdmin(ctx, orgID) {             return fmt.Errorf("untrusted skills require admin approval")         }         // 内容审查         if err := s.validator.Validate(skill.Content); err != nil {             return fmt.Errorf("skill validation failed: %w", err)         }     case "community":         // 用户需确认安装来源         if err := s.validator.Validate(skill.Content); err != nil {             return fmt.Errorf("skill validation failed: %w", err)         }         // 签名验证         if err := s.signer.Verify(skill.Content, skill.Signature, skill.CreatorID); err != nil {             return fmt.Errorf("skill signature invalid: %w", err)         }     case "verified":         if err := s.signer.Verify(skill.Content, skill.Signature, skill.CreatorID); err != nil {             return fmt.Errorf("skill signature invalid: %w", err)         }     case "trusted":         // 组织内部 Skill，直接放行     }      return s.repo.Install(orgID, skillID) } ```  #### Skill 注入 AI Prompt 时的防护  ```go // 注入前做沙箱化处理，防止 prompt injection func sanitizeSkillForPrompt(skillContent string) string {     var sb strings.Builder      // 将 Skill 内容包裹在明确的边界标记中     sb.WriteString("<skill_instructions>\n")     sb.WriteString("The following are mandatory coding guidelines. Follow them strictly.\n\n")     sb.WriteString(skillContent)     sb.WriteString("\n</skill_instructions>\n\n")     sb.WriteString("<user_task>\n")      return sb.String() } ```  关键防护措施：  | 措施 | 说明 | |------|------| | 内容审查 | 禁止 Skill 包含 shell 命令、代码执行语句、SQL 注入模式 | | 大小限制 | 单 Skill 最大 64KB，防止资源耗尽 | | 签名验证 | 公共 Skill 安装时验证 Ed25519 签名 | | 信任等级 | 四级信任体系，低信任 Skill 需管理员审核 | | Prompt 隔离 | Skill 内容用 XML 标签包裹，防止 prompt injection 越权 | | 组织中隔离 | 安装的 Skill 仅在组织范围内生效，不跨组织泄漏 |
