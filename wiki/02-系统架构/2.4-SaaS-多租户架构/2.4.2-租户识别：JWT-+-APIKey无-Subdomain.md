JWT（Web 登录用户，claims 中携带当前 orgId）         if claims := GetJWTClaims(c); claims != nil {             orgID = claims.OrgID         }          // 2. API Key（CLI / CI 调用，key 本身绑定到 org）         if apiKey := c.GetHeader("X-API-Key"); apiKey != "" {             token, err := findApiKey(c, apiKey)             if err == nil {                 orgID = token.OrgID             }         }          // 3. 组织切换：用户属于多个组织时，通过 Header 切换         if switchOrg := c.GetHeader("X-FlowCode-Org-Id"); switchOrg != "" {             if isMember(c, switchOrg) {                 orgID = switchOrg             }         }          if orgID == "" {             c.AbortWithStatusJSON(401, gin.H{"error": "tenant not identified"})             return         }          c.Set("orgID", orgID)         c.Next()     } } ```  **识别优先级**：`X-FlowCode-Org-Id`（显式切换）> JWT claims > API Key
