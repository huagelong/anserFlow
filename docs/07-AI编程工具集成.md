# 07 - AI 编程工具集成

## 7.1 概述

flowcode 通过标准 `AIToolAdapter` 接口集成主流 AI 编程工具。所有工具在 **Docker 容器沙盒**中执行，确保 AI 生成的任意代码（包括 `curl | sh`、`npm install`、`rm -rf`）完全与宿主机隔离。

> **决策**：统一使用 Docker，不引入 Bubblewrap/nsjail 等多方案。Docker 在分布式部署、资源限制、镜像缓存、日志驱动方面生态最成熟。

## 7.2 Docker 沙盒执行环境

### 架构

```
┌─────────────────────────────────────────────────────────┐
│                   flowcode Server (Go)                  │
│  ┌───────────────────────────────────────────────────┐  │
│  │              SandboxManager                       │  │
│  │                                                  │  │
│  │  ┌────────────────┐  ┌────────────────┐          │  │
│  │  │ Container A    │  │ Container B    │  ...     │  │
│  │  │ (issue-abc123) │  │ (issue-def456) │          │  │
│  │  │                │  │                │          │  │
│  │  │ flowcode/      │  │ flowcode/      │          │  │
│  │  │ sandbox:latest │  │ sandbox:latest │          │  │
│  │  └────────────────┘  └────────────────┘          │  │
│  └──────────────┬────────────────────────────────────┘  │
│                 │ Docker SDK (containerd)               │
└─────────────────┼──────────────────────────────────────┘
                  │
         ┌────────┴────────┐
         │    dockerd      │
         │  (Unix Socket)  │
         └─────────────────┘
```

每个 Issue 执行时，`SandboxManager`：

1. 拉取/复用 `flowcode/sandbox:latest` 镜像（预装 Git + Node/Python/Go + opencode/codex/claude-code CLI）
2. 创建容器，挂载独立 volume `flowcode-sandbox-{issueId}` 作为 `/workspace`
3. 容器内 `git clone` 目标仓库到 `/workspace/repo`
4. 注入 Skills 内容到 `/workspace/.flowcode_prompt.md`
5. 容器内通过 CLI 调用 AI 工具（`opencode run --workspace /workspace/repo ...`）
6. 实时捕获 stdout/stderr → WebSocket 推送
7. 执行完毕：提交变更 → push → 销毁容器（`KeepOnError` 时保留现场）

### Docker 沙盒镜像

#### 基础镜像（Go + Node + Python）

```dockerfile
FROM ubuntu:22.04
RUN apt-get update && apt-get install -y --no-install-recommends \
    git curl wget ca-certificates openjdk-21-jdk maven \
    nodejs npm python3 python3-pip golang-go \
```

> **原理**：TestCase.framework 决定执行命令，项目级 sandbox 镜像决定运行环境。两者解耦——同一框架（如 pytest）可运行在任何预装了 Python 的镜像中。

### SandboxManager 实现

```go
// internal/service/sandbox/docker.go

import (
    "github.com/docker/docker/client"
    "github.com/docker/docker/api/types"
    "github.com/docker/docker/api/types/container"
    "github.com/docker/docker/api/types/mount"
)

type DockerSandbox struct {
    cli     *client.Client
    image   string               // "flowcode/sandbox:latest"
    network string               // 容器网络模式
}

func NewDockerSandbox(cli *client.Client) *DockerSandbox {
    return &DockerSandbox{
        cli:     cli,
        image:   os.Getenv("SANDBOX_IMAGE"),  // 默认 flowcode/sandbox:latest
        network: "bridge",
    }
}

// Create 创建沙盒容器（不启动）
func (d *DockerSandbox) Create(ctx context.Context, id string, cfg SandboxConfig) (*Sandbox, error) {
    volumeName := fmt.Sprintf("flowcode-sandbox-%s", id)

    resp, err := d.cli.ContainerCreate(ctx,
        &container.Config{
            Image:      d.image,
            Cmd:        []string{"sleep", "infinity"},  // 保持运行
            WorkingDir: "/workspace",
            Env:        cfg.Env,
        },
        &container.HostConfig{
            Mounts: []mount.Mount{
                {
                    Type:   mount.TypeVolume,
                    Source: volumeName,
                    Target: "/workspace",
                },
            },
            Resources: container.Resources{
                Memory:   cfg.MemoryMB * 1024 * 1024,   // 2GB
                NanoCPUs: int64(cfg.CPUs * 1e9),        // 2 CPUs
                DiskQuota: cfg.MaxDiskMB * 1024 * 1024,  // 2GB disk
            },
            NetworkMode: container.NetworkMode(d.network),
            AutoRemove:  !cfg.KeepOnError,  // 正常自动删除
        },
        nil, nil, id,
    )
    if err != nil {
        return nil, fmt.Errorf("create container: %w", err)
    }

    return &Sandbox{
        ID:          id,
        ContainerID: resp.ID,
        VolumeName:  volumeName,
        WorkDir:     "/workspace",
    }, nil
}

// Exec 在沙盒容器内执行命令，流式收集日志
func (d *DockerSandbox) Exec(ctx context.Context, sb *Sandbox, cmd []string, onLog func(string)) (*ExecResult, error) {
    execConfig := types.ExecConfig{
        Cmd:          cmd,
        AttachStdout: true,
        AttachStderr: true,
        WorkingDir:   sb.WorkDir,
    }

    execResp, err := d.cli.ContainerExecCreate(ctx, sb.ContainerID, execConfig)
    if err != nil {
        return nil, fmt.Errorf("exec create: %w", err)
    }

    attachResp, err := d.cli.ContainerExecAttach(ctx, execResp.ID, types.ExecStartCheck{})
    if err != nil {
        return nil, fmt.Errorf("exec attach: %w", err)
    }
    defer attachResp.Close()

    // 流式读取
    scanner := bufio.NewScanner(attachResp.Reader)
    var fullOutput strings.Builder
    for scanner.Scan() {
        line := scanner.Text()
        fullOutput.WriteString(line + "\n")
        if onLog != nil {
            onLog(line)
        }
    }

    // 获取退出码
    inspectResp, _ := d.cli.ContainerExecInspect(ctx, execResp.ID)

    return &ExecResult{
        Output:   fullOutput.String(),
        ExitCode: inspectResp.ExitCode,
    }, nil
}

// Cleanup 清理沙盒
func (d *DockerSandbox) Cleanup(ctx context.Context, sb *Sandbox) error {
    // 停止容器
    timeout := 10
    d.cli.ContainerStop(ctx, sb.ContainerID, container.StopOptions{Timeout: &timeout})
    // 删除卷
    d.cli.VolumeRemove(ctx, sb.VolumeName, true)
    return nil
}
```

```go
// internal/service/sandbox/config.go

type SandboxGlobalConfig struct {
    Image          string            `yaml:"image"`
    MemoryMB       int64             `yaml:"memory_mb"`
    CPUs           int64             `yaml:"cpus"`
    DiskMB         int64             `yaml:"disk_mb"`
    TimeoutSeconds int               `yaml:"timeout_seconds"`
    KeepOnError    bool              `yaml:"keep_on_error"`
    Network        string            `yaml:"network"`
    Nodes          []string          `yaml:"nodes"`
    Schedule       string            `yaml:"schedule"`
    Overrides      map[string]OverrideConfig `yaml:"overrides"`
}

type OverrideConfig struct {
    Image    *string `yaml:"image"`
    MemoryMB *int64  `yaml:"memory_mb"`
    CPUs     *int64  `yaml:"cpus"`
    DiskMB   *int64  `yaml:"disk_mb"`
}

// Resolve 合并全局 + 项目覆盖
func (c *SandboxGlobalConfig) Resolve(projectID string) SandboxConfig {
    cfg := SandboxConfig{
        Image:    c.Image,
        MemoryMB: c.MemoryMB,
        CPUs:     c.CPUs,
        DiskMB:   c.DiskMB,
    }
    if ov, ok := c.Overrides[projectID]; ok {
        if ov.Image != nil    { cfg.Image = *ov.Image }
        if ov.MemoryMB != nil { cfg.MemoryMB = *ov.MemoryMB }
        if ov.CPUs != nil     { cfg.CPUs = *ov.CPUs }
        if ov.DiskMB != nil   { cfg.DiskMB = *ov.DiskMB }
    }
    return cfg
}
```

### 多节点 Docker 调度

当单机资源不足，在 `config/sandbox.yaml` 的 `nodes` 中添加远程节点即可，无需改代码：

```yaml
nodes:
  - "unix:///var/run/docker.sock"
  - "tcp://sandbox-node1:2375"
  - "tcp://sandbox-node2:2375"
```

```go
// 三种调度策略
func newScheduler(strategy string) Scheduler {
    switch strategy {
    case "round_robin":
        return &RoundRobinScheduler{next: 0}
    case "least_containers":
        return &LeastContainersScheduler{}   // 默认：选容器数最少的节点
    default:
        return &RandomScheduler{}            // 随机
    }
}

// 最少容器数策略
func (s *LeastContainersScheduler) Select(clients []*client.Client) *client.Client {
    var best *client.Client
    minCount := int(^uint(0) >> 1)
    for _, cli := range clients {
        info, _ := cli.Info(context.Background())
        if info.ContainersRunning < minCount {
            minCount = info.ContainersRunning
            best = cli
        }
    }
    return best
}
```

### 与 Git Clone 的衔接

```go
func (d *DockerSandbox) PrepareWorkspace(ctx context.Context, sb *Sandbox, project *model.Project) error {
    // 在容器内 clone 代码
    cloneCmd := []string{"git", "clone", "--depth", "1", project.GitRepoUrl, "/workspace/repo"}
    if _, err := d.Exec(ctx, sb, cloneCmd, nil); err != nil {
        return fmt.Errorf("clone in sandbox: %w", err)
    }

    // 注入 Skills
    skillsContent := d.loadSkills(ctx, project.ID)
    writeFileCmd := fmt.Sprintf("echo '%s' > /workspace/.flowcode_prompt.md", skillsContent)
    d.Exec(ctx, sb, []string{"sh", "-c", writeFileCmd}, nil)

    return nil
}

// 销毁沙盒前提交变更
func (d *DockerSandbox) CommitAndPush(ctx context.Context, sb *Sandbox, issue *model.Issue) (string, error) {
    cmds := [][]string{
        {"git", "-C", "/workspace/repo", "add", "-A"},
        {"git", "-C", "/workspace/repo", "commit", "-m", issue.Title},
        {"git", "-C", "/workspace/repo", "push", "origin", issue.GitBranch},
    }
    var lastSHA string
    for _, cmd := range cmds {
        result, err := d.Exec(ctx, sb, cmd, nil)
        if err != nil {
            return "", err
        }
        if strings.Contains(strings.Join(cmd, " "), "push") {
            lastSHA = extractSHA(result.Output)
        }
    }
    return lastSHA, nil
}
```

```go
nodes := []string{

func selectNode(nodes []string) *client.Client {
    // 按 CPU/内存负载选择最空闲节点
    for _, n := range nodes {
        cli, _ := client.NewClientWithOpts(client.WithHost(n))
        info, _ := cli.Info(context.Background())
        if info.ContainersRunning < 10 {  // 简单策略
            return cli
        }
    }
    return nil
}
```

### 工具概述

```go
// internal/adapter/ai_tool/opencode.go

type OpenCodeAdapter struct {
    BaseAIToolAdapterImpl
    sandbox *SandboxManager
}

func (a *OpenCodeAdapter) Execute(ctx ExecuteContext) (*ExecuteResult, error) {
    startTime := time.Now()
    
    // 1. 准备 prompt 文件 (注入 Skills)
    promptContent := buildPromptWithSkills(ctx.Prompt, ctx.ProjectCtx.SkillContents)
    promptFile := filepath.Join(ctx.WorkspaceDir, ".flowcode_prompt.md")
    os.WriteFile(promptFile, []byte(promptContent), 0644)

    // 2. 通过本地进程调用 opencode CLI
    cmdCtx, cancel := context.WithTimeout(ctx.Ctx, 10*time.Minute)
    defer cancel()

    cmd := exec.CommandContext(cmdCtx, "opencode",
        "run",
        "--prompt-file", promptFile,
        "--workspace", ctx.WorkspaceDir,
        "--model", getString(ctx.Config, "model", "claude-sonnet-4-20250514"),
        "--max-turns", "50",
    )
    cmd.Dir = ctx.WorkspaceDir

    // 3. 流式收集日志
    stdoutPipe, _ := cmd.StdoutPipe()
    stderrPipe, _ := cmd.StderrPipe()
    
    var stdout, stderr strings.Builder
    go streamLogs(stdoutPipe, &stdout, ctx.OnLog, "info")
    go streamLogs(stderrPipe, &stderr, func(e LogEntry) {
        e.Level = "warn"
        ctx.OnLog(e)
    }, "warn")

    // 4. 执行
    err := cmd.Start()
    if err != nil {
        return nil, fmt.Errorf("start opencode: %w", err)
    }
    err = cmd.Wait()

    exitCode := 0
    if err != nil {
        if exitErr, ok := err.(*exec.ExitError); ok {
            exitCode = exitErr.ExitCode()
        }
    }

    // 5. 收集变更文件列表
    filesChanged := getGitChangedFiles(ctx.WorkspaceDir)

    return &ExecuteResult{
        Success:      exitCode == 0,
        ExitCode:     exitCode,
        FilesChanged: filesChanged,
        Stdout:       stdout.String(),
        Stderr:       stderr.String(),
        Duration:     time.Since(startTime).Milliseconds(),
    }, nil
}
```

### 适配方式

```go
// internal/adapter/ai_tool/codex.go

type CodexAdapter struct {
    BaseAIToolAdapterImpl
    sandbox *SandboxManager
}

func (a *CodexAdapter) Execute(ctx ExecuteContext) (*ExecuteResult, error) {
    startTime := time.Now()

    // 准备 prompt 文件
    promptFile := filepath.Join(ctx.WorkspaceDir, ".flowcode_prompt.md")
    os.WriteFile(promptFile, []byte(ctx.Prompt), 0644)

    // 通过本地进程调用 codex CLI
    cmdCtx, cancel := context.WithTimeout(ctx.Ctx, 10*time.Minute)
    defer cancel()

    model := getString(ctx.Config, "model", "gpt-5")
    endpoint := getString(ctx.Config, "endpoint", "https://api.openai.com/v1")
    apiKey := getString(ctx.Config, "apiKey", "")

    cmd := exec.CommandContext(cmdCtx, "codex",
        "exec",
        "--prompt-file", promptFile,
        "--model", model,
        "--approval-mode", "auto",
        "--output-format", "stream-json",
    )
    cmd.Dir = ctx.WorkspaceDir
    cmd.Env = append(os.Environ(),
        "OPENAI_API_KEY="+apiKey,
        "OPENAI_BASE_URL="+endpoint,
    )

    // 流式收集日志
    stdoutPipe, _ := cmd.StdoutPipe()
    stderrPipe, _ := cmd.StderrPipe()

    var stdout, stderr strings.Builder
    var lastProgress int
    var mu sync.Mutex

    go scanStreamJSON(stdoutPipe, func(line []byte) {
        mu.Lock()
        defer mu.Unlock()
        var msg map[string]any
        if err := json.Unmarshal(line, &msg); err != nil {
            ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "debug", Content: string(line)})
            return
        }
        stdout.Write(line)
        stdout.WriteString("\n")

        switch msg["type"] {
        case "assistant":
            content := msg["message"].(map[string]any)["content"].(string)
            ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "info", Content: content})
        case "tool_use":
            ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "info",
                Content: fmt.Sprintf("[Tool] %v: %v", msg["name"], msg["input"]),
                Step:    fmt.Sprint(msg["name"])})
        case "progress":
            pct := int(msg["percent"].(float64))
            lastProgress = pct
            ctx.OnProgress(pct, fmt.Sprint(msg["detail"]))
        }
    })

    go scanStreamText(stderrPipe, func(line string) {
        ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "warn", Content: line})
        stderr.WriteString(line + "\n")
    })

    // 执行
    err := cmd.Start()
    if err != nil {
        return nil, fmt.Errorf("start codex: %w", err)
    }
    err = cmd.Wait()

    exitCode := 0
    if err != nil {
        if exitErr, ok := err.(*exec.ExitError); ok {
            exitCode = exitErr.ExitCode()
        }
    }

    filesChanged := getGitChangedFiles(ctx.WorkspaceDir)

    return &ExecuteResult{
        Success:      exitCode == 0,
        ExitCode:     exitCode,
        FilesChanged: filesChanged,
        Stdout:       stdout.String(),
        Stderr:       stderr.String(),
        Duration:     time.Since(startTime).Milliseconds(),
        Metadata:     map[string]any{"lastProgress": lastProgress},
    }, nil
}
```

```go
    stderrPipe, _ := cmd.StderrPipe()

    var stdout, stderr strings.Builder

    go scanStreamJSON(stdoutPipe, func(line []byte) {
        var msg map[string]any
        if err := json.Unmarshal(line, &msg); err != nil {
            ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "debug", Content: string(line)})
            return
        }
        stdout.Write(line)
        stdout.WriteString("\n")

        switch msg["type"] {
        case "stream_event":
            if event, ok := msg["event"].(map[string]any); ok {
                if cb, ok := event["content_block"].(map[string]any); ok {
                    if text, ok := cb["text"].(string); ok {
                        ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "info", Content: text})
                    }
                }
            }
        case "tool_result":
            ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "info",
                Content: fmt.Sprintf("[Tool Result] %v: OK", msg["tool_name"]),
                Step:    fmt.Sprint(msg["tool_name"])})
        case "error":
            ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "error", Content: fmt.Sprint(msg["message"])})
        }
    })

    go scanStreamText(stderrPipe, func(line string) {
        ctx.OnLog(LogEntry{Timestamp: time.Now().UnixMilli(), Level: "warn", Content: line})
        stderr.WriteString(line + "\n")
    })

    // 执行
    err := cmd.Start()
    if err != nil {
        return nil, fmt.Errorf("start claude: %w", err)
    }
    err = cmd.Wait()

    exitCode := 0
    if err != nil {
        if exitErr, ok := err.(*exec.ExitError); ok {
            exitCode = exitErr.ExitCode()
        }
    }

    filesChanged := getGitChangedFiles(ctx.WorkspaceDir)

    return &ExecuteResult{
        Success:      exitCode == 0,
        ExitCode:     exitCode,
        FilesChanged: filesChanged,
        Stdout:       stdout.String(),
        Stderr:       stderr.String(),
        Duration:     time.Since(startTime).Milliseconds(),
    }, nil
}
```

## 7.6b 自动分类适配器 (ClassifyAdapter)

未命中关键词的 Issue 通过 opencode 异步补充分类。与编码执行不同，分类任务仅一次 LLM 调用，无需克隆代码，直接在 prompt 中传递标题和描述。

### 适配器实现

```go
// internal/adapter/ai_tool/classify.go

type ClassifyAdapter struct {
    sandbox *SandboxManager
}

// ClassifyInput 分类请求
type ClassifyInput struct {
    Title       string `json:"title"`
    Description string `json:"description"`
    ProjectName string `json:"project_name"` // 提供上下文
}

// ClassifyResult AI 分类结果
type ClassifyResult struct {
    Category   string  `json:"category"`   // feature/bug/refactor/docs/infra/test
    Confidence float64 `json:"confidence"` // 0.0 ~ 1.0
    Reason     string  `json:"reason"`     // 一句话解释
}

func (a *ClassifyAdapter) Classify(ctx context.Context, input ClassifyInput) (*ClassifyResult, error) {
    // 1. 构建分类 prompt
    prompt := fmt.Sprintf(`你是一个 Issue 分类器。根据标题和描述将 Issue 分到以下类别之一：
- feature: 新功能、新增特性
- bug: 错误修复、异常处理
- refactor: 代码重构、性能优化
- docs: 文档、注释
- infra: 部署、CI/CD、环境配置、依赖升级
- test: 测试用例、测试相关

输出严格 JSON 格式：{"category":"...","confidence":0.95,"reason":"..."}

项目: %s
标题: %s
描述: %s`, input.ProjectName, input.Title, truncate(input.Description, 2000))

    // 2. 写入临时 prompt 文件
    promptFile := filepath.Join(a.sandbox.BasePath, "classify", uuid.New().String()+".md")
    os.MkdirAll(filepath.Dir(promptFile), 0755)
    os.WriteFile(promptFile, []byte(prompt), 0644)
    defer os.Remove(promptFile)

    // 3. 调用 opencode (单次调用，不进入交互模式)
    ctx, cancel := context.WithTimeout(ctx, 30*time.Second)
    defer cancel()

    cmd := exec.CommandContext(ctx, "opencode",
        "run",
        "--prompt-file", promptFile,
        "--model", "claude-haiku",        // 分类用轻量模型
        "--max-turns", "1",               // 单轮即可
        "--output-format", "json",
    )

    output, err := cmd.CombinedOutput()
    if err != nil {
        return nil, fmt.Errorf("classify failed: %w, output: %s", err, string(output))
    }

    // 4. 解析 JSON 结果
    var result ClassifyResult
    if err := json.Unmarshal(extractJSON(string(output)), &result); err != nil {
        return nil, fmt.Errorf("parse classify result: %w", err)
    }

    // 5. 校验分类值
    validCategories := map[string]bool{
        "feature": true, "bug": true, "refactor": true,
        "docs": true, "infra": true, "test": true,
    }
    if !validCategories[result.Category] {
        return nil, fmt.Errorf("invalid category: %s", result.Category)
    }

    return &result, nil
}
```

### 异步分类任务

```go
// internal/queue/classify_handler.go

func HandleClassifyTask(ctx context.Context, t *asynq.Task) error {
    var payload ClassifyPayload
    json.Unmarshal(t.Payload(), &payload)

    // 1. 加载 Issue
    var issue model.Issue
    db.First(&issue, "id = ?", payload.IssueID)

    // 2. 加载项目（获取上下文）
    var project model.Project
    db.First(&project, "id = ?", issue.ProjectID)

    // 3. 调用 ClassifyAdapter
    adapter := &ClassifyAdapter{sandbox: sandboxMgr}
    result, err := adapter.Classify(ctx, ClassifyInput{
        Title:       issue.Title,
        Description: issue.Description,
        ProjectName: project.Name,
    })
    if err != nil {
        // 分类失败不影响 Issue 创建，保持规则分类结果
        slog.Warn("AI classify failed, keeping rule result", "issue", issue.ID, "err", err)
        return nil // 非致命错误，不重试
    }

    // 4. 回写分类结果
    db.Model(&issue).Updates(map[string]any{
        "category":             result.Category,
        "category_source":      "ai",
        "category_confidence":  result.Confidence,
        "category_classified_at": time.Now(),
    })

    // 5. 发射分类完成事件 (前端可据此刷新)
    hookRegistry.Emit(ctx, "issue:classified", map[string]any{
        "issue_id":    issue.ID,
        "category":    result.Category,
        "confidence":  result.Confidence,
        "source":      "ai",
        "reason":      result.Reason,
    })

    return nil
}
```

```go
// internal/service/test_runner.go

type TestRunner struct {
    db       *gorm.DB
    aiTool   adapter.AIToolAdapter
    sandbox  *SandboxManager
}

// Run 在 Docker 沙盒中执行测试代码，返回结构化结果
func (r *TestRunner) Run(ctx context.Context, tc *model.TestCase) (*TestResult, error) {
    issue, _ := r.getIssue(ctx, tc.IssueID)
    project, _ := r.getProject(ctx, issue.ProjectID)

    // 1. 创建 Docker 沙盒（项目级可覆盖镜像，见 config/sandbox.yaml overrides）
    sb, err := r.sandbox.Create(ctx, project.OrgID+"/test-"+tc.ID)
    if err != nil {
        return nil, fmt.Errorf("create sandbox: %w", err)
    }
    defer r.sandbox.Cleanup(ctx, sb)

    // 2. 拉取项目代码
    if err := r.sandbox.PrepareWorkspace(ctx, sb, project); err != nil {
        return nil, fmt.Errorf("prepare workspace: %w", err)
    }

    // 3. 安装依赖（npm install / pip install / go mod download 等）
    if depsCmd := r.getDepsInstallCmd(tc.Framework); depsCmd != nil {
        r.sandbox.Exec(ctx, sb, depsCmd, nil) // 失败不阻断，可能依赖已就绪
    }

    // 4. 注入测试代码到容器内正确位置
    testPath := fmt.Sprintf("/workspace/repo/%s", tc.TestFilePath)
    writeCmd := []string{"sh", "-c", fmt.Sprintf(
        "cat > %s << 'EOTEST'\n%s\nEOTEST", testPath, tc.TestCode,
    )}
    if _, err := r.sandbox.Exec(ctx, sb, writeCmd, nil); err != nil {
        return nil, fmt.Errorf("inject test code: %w", err)
    }

    // 5. 执行测试命令
    cmd := r.buildTestCmd(tc.Framework)
    result, err := r.sandbox.Exec(ctx, sb, cmd, nil)
    if err != nil {
        return nil, fmt.Errorf("run test: %w", err)
    }

    // 6. 解析结构化结果
    return r.parseResult(result.Output, tc.Framework), nil
}

// ── 语言/框架 → 依赖安装 + 测试命令映射 ──

func (r *TestRunner) getDepsInstallCmd(framework string) []string {
    switch framework {
    case "testing":
        return []string{"sh", "-c", "cd /workspace/repo && go mod download 2>/dev/null; true"}
    case "vitest", "jest", "mocha":
        return []string{"sh", "-c", "cd /workspace/repo && npm install 2>/dev/null; true"}
    case "pytest", "unittest":
        return []string{"sh", "-c", "cd /workspace/repo && pip install -r requirements.txt 2>/dev/null; true"}
    case "maven":
        return []string{"sh", "-c", "cd /workspace/repo && mvn dependency:resolve -q 2>/dev/null; true"}
    case "dotnet":
        return []string{"sh", "-c", "cd /workspace/repo && dotnet restore 2>/dev/null; true"}
    default:
        return nil // gradle / cargo / 其他不需要显式安装
    }
}

func (r *TestRunner) buildTestCmd(framework string) []string {
    switch framework {
    // Go
    case "testing":
        return []string{"sh", "-c", "cd /workspace/repo && go test ./... -v -cover 2>&1"}

    // TypeScript / JavaScript
    case "vitest":
        return []string{"sh", "-c", "cd /workspace/repo && npx vitest run --reporter=verbose 2>&1"}
    case "jest":
        return []string{"sh", "-c", "cd /workspace/repo && npx jest --verbose 2>&1"}
    case "mocha":
        return []string{"sh", "-c", "cd /workspace/repo && npx mocha --reporter=spec 2>&1"}

    // Python
    case "pytest":
        return []string{"sh", "-c", "cd /workspace/repo && python -m pytest -v --tb=short 2>&1"}
    case "unittest":
        return []string{"sh", "-c", "cd /workspace/repo && python -m unittest discover -v 2>&1"}

    // Java
    case "maven":
        return []string{"sh", "-c", "cd /workspace/repo && mvn test -B 2>&1"}
    case "gradle":
        return []string{"sh", "-c", "cd /workspace/repo && ./gradlew test --console=plain 2>&1"}

    // Rust
    case "cargo":
        return []string{"sh", "-c", "cd /workspace/repo && cargo test --verbose 2>&1"}

    // .NET
    case "dotnet":
        return []string{"sh", "-c", "cd /workspace/repo && dotnet test --verbosity normal 2>&1"}

    default:
        return []string{"sh", "-c", fmt.Sprintf("cd /workspace/repo && %s 2>&1", framework)}
    }
}

func (r *TestRunner) parseResult(output string, framework string) *TestResult {
    tr := &TestResult{
        Output: output,
        RunAt:  time.Now(),
    }

    // 通用解析：匹配 PASS/FAIL 行和覆盖率
    passRe := regexp.MustCompile(`--- PASS:`)
    failRe := regexp.MustCompile(`--- FAIL:`)
    covRe := regexp.MustCompile(`coverage: ([\d.]+)%`)

    tr.Passed = len(passRe.FindAllString(output, -1))
    tr.Failed = len(failRe.FindAllString(output, -1))
    tr.Total = tr.Passed + tr.Failed

    if matches := covRe.FindStringSubmatch(output); len(matches) > 1 {
        tr.Coverage = matches[1] + "%"
    }

    if tr.Failed == 0 && tr.Total > 0 {
        tr.Status = "passed"
    } else {
        tr.Status = "failed"
        // 提取失败详情
        tr.Errors = r.extractFailureDetails(output)
    }

    return tr
}

func (r *TestRunner) extractFailureDetails(output string) []TestError {
    re := regexp.MustCompile(`(\S+):(\d+):\s*(.+)`)
    var errors []TestError
    for _, match := range re.FindAllStringSubmatch(output, -1) {
        line, _ := strconv.Atoi(match[2])
        errors = append(errors, TestError{
            File:    match[1],
            Line:    line,
            Message: match[3],
        })
    }
    return errors
}
```
